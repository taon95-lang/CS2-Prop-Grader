import requests
from bs4 import BeautifulSoup
import re

def fetch_player_map_stats(player_id, player_name, api_key):
    """
    Targets the HLTV 'Matches' tab for a player.
    Extracts Kills, Headshots, and Rounds for Maps 1-2 of the last 10 BO3s.
    """
    # ScraperAPI URL construction
    target_url = f"https://www.hltv.org/stats/players/matches/{player_id}/{player_name}"
    proxy_url = f"http://api.scraperapi.com?api_key={api_key}&url={target_url}"
    
    print(f"[*] Accessing HLTV for {player_name} via ScraperAPI...")
    response = requests.get(proxy_url)
    
    if response.status_code != 200:
        print(f"[!] Error: ScraperAPI returned status {response.status_code}")
        return []

    soup = BeautifulSoup(response.content, 'html.parser')
    
    # 1. Find the stats table
    table = soup.find('table', {'class': 'stats-table'})
    if not table:
        print("[!] Could not find stats table.")
        return []

    rows = table.find('tbody').find_all('tr')
    
    all_maps = []
    
    # HLTV lists matches chronologically. We iterate to find BO3s.
    # Note: We need to group rows by match date/opponent to identify BO3s.
    for row in rows:
        cols = row.find_all('td')
        if len(cols) < 5: continue
        
        # Extract Map Name
        map_name = cols[1].text.strip()
        
        # Extract Kills and HS: Format is usually "21 (11)"
        k_hs_text = cols[4].text.strip()
        # Regex to pull '21' as kills and '11' as headshots
        match = re.search(r'(\d+)\s*\((\d+)\)', k_hs_text)
        
        if match:
            kills = int(match.group(1))
            headshots = int(match.group(2))
        else:
            # Fallback if HS isn't in parens
            kills = int(re.search(r'(\d+)', k_hs_text).group(1))
            headshots = 0

        # Extract Rounds
        # HLTV format for result is "16 - 12" or "13 - 10"
        result_text = cols[3].text.strip()
        rounds = sum(map(int, re.findall(r'\d+', result_text)))

        all_maps.append({
            'map': map_name,
            'kills': kills,
            'hs': headshots,
            'rounds': rounds
        })

    # --- THE FILTERING LOGIC ---
    # To follow your 'Gold Standard', we group by Match and take only Maps 1-2
    # For simplicity in this script, we assume the list is chronological.
    # We take the first 20 maps (10 series * 2 maps each)
    gold_standard_data = all_maps[:20] 
    
    return gold_standard_data
